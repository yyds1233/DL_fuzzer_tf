import atheris

with atheris.instrument_imports():
    import sys
    import tensorflow as tf


def _consume_dim(fdp):
    return fdp.ConsumeIntInRange(0, 8)


def _consume_rank(fdp):
    mode = fdp.ConsumeIntInRange(0, 6)
    if mode == 0:
        return 1
    if mode == 1:
        return 2
    if mode == 2:
        return 3
    if mode == 3:
        return 4
    if mode == 4:
        return 0
    return fdp.ConsumeIntInRange(0, 5)


def _consume_shape(fdp):
    rank = _consume_rank(fdp)
    return [_consume_dim(fdp) for _ in range(rank)]


def _num_elements(shape):
    n = 1
    for d in shape:
        n *= d
    return n


def _make_bool_tensor(fdp, shape):
    count = _num_elements(shape)
    values = [fdp.ConsumeBool() for _ in range(count)]
    return tf.constant(values, dtype=tf.bool, shape=shape)


def _broadcastable_shape_from(fdp, base_shape):
    rank = len(base_shape)
    strategy = fdp.ConsumeIntInRange(0, 6)

    if strategy == 0:
        return list(base_shape)

    if strategy == 1:
        return [1] * rank

    if strategy == 2:
        result = []
        for d in base_shape:
            result.append(d if fdp.ConsumeBool() else 1)
        return result

    if strategy == 3:
        if rank == 0:
            return []
        k = fdp.ConsumeIntInRange(0, rank)
        return list(base_shape[rank - k:])

    if strategy == 4:
        if rank == 0:
            return []
        prefix = [1] * fdp.ConsumeIntInRange(0, 3)
        result = []
        for d in base_shape:
            result.append(d if fdp.ConsumeBool() else 1)
        return prefix + result

    if strategy == 5:
        if rank == 0:
            return []
        result = list(base_shape)
        idx = fdp.ConsumeIntInRange(0, rank - 1)
        result[idx] = 1
        return result

    return _consume_shape(fdp)


@atheris.instrument_func
def TestOneInput(data: bytes):
    fdp = atheris.FuzzedDataProvider(data)
    try:
        if fdp.ConsumeBool():
            tf.config.run_functions_eagerly(fdp.ConsumeBool())

        x_shape = _consume_shape(fdp)
        if fdp.ConsumeBool():
            y_shape = _broadcastable_shape_from(fdp, x_shape)
        else:
            y_shape = _consume_shape(fdp)

        x = _make_bool_tensor(fdp, x_shape)
        y = _make_bool_tensor(fdp, y_shape)

        name = None
        if fdp.ConsumeBool():
            name = fdp.ConsumeUnicodeNoSurrogates(16)

        variant = fdp.ConsumeIntInRange(0, 2)
        if variant == 0:
            result = tf.math.logical_or(x, y, name=name)
        elif variant == 1:
            result = tf.raw_ops.LogicalOr(x=x, y=y, name=name)
        else:
            @tf.function
            def _fn(a, b):
                return tf.math.logical_or(a, b, name=name)
            result = _fn(x, y)

        if fdp.ConsumeBool():
            _ = tf.identity(result)
        if fdp.ConsumeBool():
            _ = tf.math.reduce_any(result)
        if fdp.ConsumeBool():
            _ = tf.math.reduce_all(result)
        if fdp.ConsumeBool() and tf.executing_eagerly():
            _ = result.numpy()

    except (
        tf.errors.InvalidArgumentError,
        tf.errors.UnimplementedError,
        tf.errors.InternalError,
        tf.errors.ResourceExhaustedError,
        ValueError,
        TypeError,
        RuntimeError,
        AssertionError,
        IndexError,
        NotImplementedError,
    ):
        return
    except Exception:
        return


def main():
    atheris.Setup(sys.argv, TestOneInput)
    atheris.Fuzz()


if __name__ == "__main__":
    main()
