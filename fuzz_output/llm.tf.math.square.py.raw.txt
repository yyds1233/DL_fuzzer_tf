import atheris

with atheris.instrument_imports():
    import sys
    import math
    import tensorflow as tf


_DTYPE_CHOICES = [
    tf.bfloat16,
    tf.float16,
    tf.float32,
    tf.float64,
    tf.int8,
    tf.int16,
    tf.int32,
    tf.int64,
    tf.uint8,
    tf.uint16,
    tf.uint32,
    tf.uint64,
    tf.complex64,
    tf.complex128,
]


def _consume_shape(fdp):
    rank = fdp.ConsumeIntInRange(1, 4)
    if rank == 1:
        return [fdp.ConsumeIntInRange(1, 8)]
    if rank == 2:
        return [
            fdp.ConsumeIntInRange(1, 16),
            fdp.ConsumeIntInRange(1, 11),
        ]
    if rank == 3:
        return [
            fdp.ConsumeIntInRange(1, 16),
            fdp.ConsumeIntInRange(1, 16),
            fdp.ConsumeIntInRange(1, 16),
        ]
    return [
        fdp.ConsumeIntInRange(1, 16),
        fdp.ConsumeIntInRange(1, 8),
        fdp.ConsumeIntInRange(1, 64),
        fdp.ConsumeIntInRange(1, 32),
    ]


def _num_elements(shape):
    n = 1
    for dim in shape:
        n *= dim
    return n


def _finite_float(v):
    if math.isnan(v):
        return 0.0
    if math.isinf(v):
        return 1.0 if v > 0 else -1.0
    return v


def _consume_float_list(fdp, count):
    values = []
    mode = fdp.ConsumeIntInRange(0, 4)
    for _ in range(count):
        if mode == 0:
            values.append(float(fdp.ConsumeIntInRange(-4, 4)))
        elif mode == 1:
            values.append(_finite_float(fdp.ConsumeFloat()))
        elif mode == 2:
            values.append(fdp.ConsumeFloatInRange(-1e3, 1e3))
        elif mode == 3:
            values.append(fdp.ConsumeFloatInRange(-1.0, 1.0))
        else:
            values.append(float(fdp.ConsumeIntInRange(-128, 128)))
    return values


def _consume_int_list(fdp, count, signed=True, bits=32):
    values = []
    if signed:
        if bits <= 8:
            lo, hi = -128, 127
        elif bits <= 16:
            lo, hi = -32768, 32767
        elif bits <= 32:
            lo, hi = -(2 ** 31), 2 ** 31 - 1
        else:
            lo, hi = -(2 ** 31), 2 ** 31 - 1
    else:
        if bits <= 8:
            lo, hi = 0, 255
        elif bits <= 16:
            lo, hi = 0, 65535
        elif bits <= 32:
            lo, hi = 0, 2 ** 16
        else:
            lo, hi = 0, 2 ** 16

    for _ in range(count):
        if fdp.ConsumeBool():
            values.append(fdp.ConsumeIntInRange(lo, hi))
        else:
            values.append(fdp.ConsumeIntInRange(-8, 8) if signed else fdp.ConsumeIntInRange(0, 8))
    return values


def _consume_tensor(fdp, dtype, shape):
    count = _num_elements(shape)

    if dtype in (tf.float16, tf.float32, tf.float64, tf.bfloat16):
        values = _consume_float_list(fdp, count)
        return tf.constant(values, dtype=dtype, shape=shape)

    if dtype == tf.int8:
        values = _consume_int_list(fdp, count, signed=True, bits=8)
        return tf.constant(values, dtype=dtype, shape=shape)

    if dtype == tf.int16:
        values = _consume_int_list(fdp, count, signed=True, bits=16)
        return tf.constant(values, dtype=dtype, shape=shape)

    if dtype == tf.int32:
        values = _consume_int_list(fdp, count, signed=True, bits=32)
        return tf.constant(values, dtype=dtype, shape=shape)

    if dtype == tf.int64:
        values = _consume_int_list(fdp, count, signed=True, bits=64)
        return tf.constant(values, dtype=dtype, shape=shape)

    if dtype == tf.uint8:
        values = _consume_int_list(fdp, count, signed=False, bits=8)
        return tf.constant(values, dtype=dtype, shape=shape)

    if dtype == tf.uint16:
        values = _consume_int_list(fdp, count, signed=False, bits=16)
        return tf.constant(values, dtype=dtype, shape=shape)

    if dtype == tf.uint32:
        values = _consume_int_list(fdp, count, signed=False, bits=32)
        return tf.constant(values, dtype=dtype, shape=shape)

    if dtype == tf.uint64:
        values = _consume_int_list(fdp, count, signed=False, bits=64)
        return tf.constant(values, dtype=dtype, shape=shape)

    if dtype in (tf.complex64, tf.complex128):
        real = _consume_float_list(fdp, count)
        imag = _consume_float_list(fdp, count)
        values = [complex(r, i) for r, i in zip(real, imag)]
        return tf.constant(values, dtype=dtype, shape=shape)

    values = _consume_float_list(fdp, count)
    return tf.constant(values, dtype=dtype, shape=shape)


@atheris.instrument_func
def TestOneInput(data: bytes):
    fdp = atheris.FuzzedDataProvider(data)
    try:
        dtype = _DTYPE_CHOICES[fdp.ConsumeIntInRange(0, len(_DTYPE_CHOICES) - 1)]
        shape = _consume_shape(fdp)
        x = _consume_tensor(fdp, dtype, shape)

        if fdp.ConsumeBool():
            result = tf.math.square(x)
        else:
            name = fdp.ConsumeUnicodeNoSurrogates(32)
            result = tf.math.square(x, name=name)

        if fdp.ConsumeBool():
            _ = tf.raw_ops.Square(x=x)

        if fdp.ConsumeBool():
            _ = tf.convert_to_tensor(result)
            _ = tf.shape(result)

        if fdp.ConsumeBool():
            _ = result.numpy()

    except (
        tf.errors.InvalidArgumentError,
        tf.errors.UnimplementedError,
        tf.errors.InternalError,
        tf.errors.ResourceExhaustedError,
        ValueError,
        TypeError,
        RuntimeError,
        AssertionError,
        IndexError,
        NotImplementedError,
    ):
        return
    except Exception:
        return


def main():
    atheris.Setup(sys.argv, TestOneInput)
    atheris.Fuzz()


if __name__ == "__main__":
    main()
