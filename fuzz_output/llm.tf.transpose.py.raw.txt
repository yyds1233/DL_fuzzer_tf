import atheris

with atheris.instrument_imports():
    import sys
    import numpy as np
    import tensorflow as tf


def _consume_dtype(fdp):
    dtypes = [
        tf.int8,
        tf.uint8,
        tf.int16,
        tf.uint16,
        tf.int32,
        tf.int64,
        tf.float16,
        tf.float32,
        tf.float64,
        tf.bool,
        tf.complex64,
        tf.complex128,
        tf.bfloat16,
        tf.string,
    ]
    return dtypes[fdp.ConsumeIntInRange(0, len(dtypes) - 1)]


def _consume_perm_dtype(fdp):
    return tf.int32 if fdp.ConsumeBool() else tf.int64


def _consume_shape(fdp):
    rank = [2, 3, 4][fdp.ConsumeIntInRange(0, 2)]
    shape = []
    for _ in range(rank):
        if fdp.ConsumeBool():
            shape.append(fdp.ConsumeIntInRange(1, 4))
        else:
            shape.append(fdp.ConsumeIntInRange(1, 16))
    return shape


def _num_elements(shape):
    total = 1
    for d in shape:
        total *= d
    return total


def _finite_float32(val):
    if np.isnan(val):
        return np.float32(0.0)
    if np.isposinf(val):
        return np.float32(np.finfo(np.float32).max)
    if np.isneginf(val):
        return np.float32(np.finfo(np.float32).min)
    return np.float32(val)


def _make_numpy_array(fdp, dtype, shape):
    total = _num_elements(shape)

    if dtype == tf.string:
        arr = []
        for _ in range(total):
            size = fdp.ConsumeIntInRange(0, 16)
            arr.append(fdp.ConsumeBytes(size))
        return np.array(arr, dtype=np.object_).reshape(shape)

    if dtype == tf.bool:
        arr = [fdp.ConsumeBool() for _ in range(total)]
        return np.array(arr, dtype=np.bool_).reshape(shape)

    if dtype in (tf.int8, tf.int16, tf.int32, tf.int64):
        info = np.iinfo(dtype.as_numpy_dtype)
        low = max(info.min, -1024)
        high = min(info.max, 1024)
        arr = [fdp.ConsumeIntInRange(low, high) for _ in range(total)]
        return np.array(arr, dtype=dtype.as_numpy_dtype).reshape(shape)

    if dtype in (tf.uint8, tf.uint16):
        info = np.iinfo(dtype.as_numpy_dtype)
        high = min(info.max, 2048)
        arr = [fdp.ConsumeIntInRange(0, high) for _ in range(total)]
        return np.array(arr, dtype=dtype.as_numpy_dtype).reshape(shape)

    if dtype in (tf.float16, tf.float32, tf.float64, tf.bfloat16):
        special = [
            -0.0,
            0.0,
            1.0,
            -1.0,
            1e-6,
            -1e-6,
            1e6,
            -1e6,
            float("inf"),
            float("-inf"),
            float("nan"),
        ]
        arr = []
        for _ in range(total):
            if fdp.ConsumeBool():
                arr.append(special[fdp.ConsumeIntInRange(0, len(special) - 1)])
            else:
                arr.append(fdp.ConsumeFloat())
        if dtype == tf.bfloat16:
            arr = [_finite_float32(x) for x in arr]
            return np.array(arr, dtype=np.float32).reshape(shape)
        return np.array(arr, dtype=dtype.as_numpy_dtype).reshape(shape)

    if dtype in (tf.complex64, tf.complex128):
        special = [0.0, 1.0, -1.0, float("inf"), float("-inf"), float("nan")]
        arr = []
        for _ in range(total):
            if fdp.ConsumeBool():
                real = special[fdp.ConsumeIntInRange(0, len(special) - 1)]
                imag = special[fdp.ConsumeIntInRange(0, len(special) - 1)]
            else:
                real = fdp.ConsumeFloat()
                imag = fdp.ConsumeFloat()
            arr.append(complex(real, imag))
        return np.array(arr, dtype=dtype.as_numpy_dtype).reshape(shape)

    raise ValueError("Unsupported dtype")


def _make_tensor(fdp, dtype, shape):
    arr = _make_numpy_array(fdp, dtype, shape)
    return tf.convert_to_tensor(arr, dtype=dtype)


def _consume_perm(fdp, rank, perm_dtype):
    mode = fdp.ConsumeIntInRange(0, 5)

    if mode == 0:
        perm = list(reversed(range(rank)))
    elif mode == 1:
        perm = list(range(rank))
    elif mode == 2:
        perm = list(range(rank))
        for i in range(rank):
            j = fdp.ConsumeIntInRange(0, rank - 1)
            perm[i], perm[j] = perm[j], perm[i]
    elif mode == 3:
        if rank == 3:
            perm = [0, 2, 1]
        else:
            perm = list(reversed(range(rank)))
    elif mode == 4:
        perm = [fdp.ConsumeIntInRange(0, rank - 1) for _ in range(rank)]
    else:
        perm_len = fdp.ConsumeIntInRange(0, rank + 2)
        perm = [fdp.ConsumeIntInRange(-2, rank + 2) for _ in range(perm_len)]

    return tf.convert_to_tensor(np.array(perm, dtype=perm_dtype.as_numpy_dtype), dtype=perm_dtype)


@atheris.instrument_func
def TestOneInput(data: bytes):
    fdp = atheris.FuzzedDataProvider(data)
    try:
        dtype = _consume_dtype(fdp)
        shape = _consume_shape(fdp)
        a = _make_tensor(fdp, dtype, shape)

        use_perm = fdp.ConsumeBool()
        use_raw = fdp.ConsumeBool()
        conjugate = fdp.ConsumeBool()
        name = None
        if fdp.ConsumeBool():
            name_len = fdp.ConsumeIntInRange(0, 24)
            name = fdp.ConsumeBytes(name_len).decode("utf-8", "ignore")

        if use_perm:
            perm_dtype = _consume_perm_dtype(fdp)
            perm = _consume_perm(fdp, len(shape), perm_dtype)
            if use_raw:
                result = tf.raw_ops.Transpose(x=a, perm=perm, name=name)
            else:
                result = tf.transpose(a=a, perm=perm, conjugate=conjugate, name=name)
        else:
            result = tf.transpose(a=a, perm=None, conjugate=conjugate, name=name)

        if fdp.ConsumeBool():
            _ = tf.shape(result)
        if fdp.ConsumeBool():
            _ = result.numpy()
        if fdp.ConsumeBool():
            _ = tf.transpose(result)
        if fdp.ConsumeBool() and result.dtype.is_complex:
            _ = tf.transpose(result, conjugate=fdp.ConsumeBool())

    except (
        tf.errors.InvalidArgumentError,
        tf.errors.UnimplementedError,
        tf.errors.InternalError,
        tf.errors.ResourceExhaustedError,
        ValueError,
        TypeError,
        RuntimeError,
        AssertionError,
        IndexError,
        NotImplementedError,
    ):
        return
    except Exception:
        return


def main():
    atheris.Setup(sys.argv, TestOneInput)
    atheris.Fuzz()


if __name__ == "__main__":
    main()
